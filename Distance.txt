@name Distance
@inputs 
@outputs 
@persist A:vector B:vector
@trigger 

if(first()) {
    A = vec(0,0,0)
    B = vec(0,0,0)
}

interval(10)

if(owner():keyPressed("v")) {
    A = owner():aimPos()
    print(A:distance(B))
}
elseif(owner():keyPressed("b")) {
    B = owner():aimPos()
    print(A:distance(B))
}
