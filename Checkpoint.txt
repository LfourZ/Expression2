@name Checkpoint
@inputs 
@outputs 
@persist X Ps:vector
@trigger 

if(first()) {
    Ps = entity():pos()
}
interval(10)

if(X==1&owner():keyPressed("v")) {
    concmd("setpos 10000 10000 10000")
    concmd("setpos " + toString(Ps:x()) + " " + toString(Ps:y()) + " " + toString(Ps:z()))
    X = 0
}
elseif(!owner():keyPressed("v")) {
    X = 1
}
